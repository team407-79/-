<!DOCTYPE html>
<html>
<head>
    <title>Hello World</title>
    <!-- 스타일 태그 내에 CSS 코드를 작성합니다 -->
    <style>
        /* body 태그에 대한 스타일을 설정합니다 */
        body {
            background-color: #f0f0f0; /* 웹페이지의 배경색을 회색으로 설정 */
            font-family: 'Arial', sans-serif; /* 웹페이지의 기본 글꼴을 Arial로 설정. Arial 글꼴이 없는 경우, sans-serif 글꼴을 사용 */
            text-align: center; /* 웹페이지의 텍스트를 가운데 정렬 */
            padding: 50px; /* 웹페이지의 내용 주위에 50픽셀의 패딩을 추가 */
        }

        /* h1 태그에 대한 스타일을 설정합니다 */
        h1 {
            color: #333; /* h1 태그의 텍스트 색상을 어두운 회색으로 설정 */
            text-shadow: 2px 2px 4px #999; /* h1 태그의 텍스트에 그림자를 추가 */
        }

        /* id가 'search-bar'인 요소에 대한 스타일을 설정합니다 */
        #search-bar {
            margin-top: 20px; /* 상단 마진을 20픽셀로 설정 */
        }
    </style>
</head>
<body>
    <!-- h1 태그를 이용하여 웹페이지의 제목을 표시합니다 -->
    <h1>Hello World!</h1>
    
    <!-- id가 'search-bar'인 div 태그 내에 검색창을 추가합니다 -->
    <div id="search-bar">
        <!-- form 태그를 이용하여 검색 폼을 생성합니다. action 속성에 Google 검색 URL을 지정하고, method 속성에는 'get'을 지정합니다 -->
        <form action="https://www.google.com/search" method="get">
            <!-- input 태그를 이용하여 검색어를 입력받는 텍스트 필드를 생성합니다. name 속성에 'q'를 지정하고, placeholder 속성에는 텍스트 필드에 표시될 힌트를 지정합니다 -->
            <input type="text" name="q" placeholder="Search...">
            <!-- input 태그를 이용하여 폼 제출 버튼을 생성합니다. value 속성에는 버튼에 표시될 텍스트를 지정합니다 -->
            <input type="submit" value="Go">
        </form>
    </div>
</body>
</html>
